
CREATE TABLE CityInfo(ZipCode int PRIMARY KEY,
		    CityName varchar(20),
		    StateCode int REFRENCES StateInfo(StateCode)
		    );

CREATE TABLE StateInfo(StateCode int PRIMARY KEY ,
	                       StateName varchar(90)
		      );
